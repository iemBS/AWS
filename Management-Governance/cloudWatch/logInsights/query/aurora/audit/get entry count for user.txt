    fields @message,@timestamp
    | filter @message like ",QUERY," 
    | parse @message "*,*,*,*,*,*,*,*,*" as ts,svrNm,usr,ip,processId,logId,eventType,db,qry

    | display usr,periodStartTs,periodEndTs,recCnt
    | sort usr
